# Install Collectd

yum -y install epel-release
yum -y install collectd
yum -y install net-tools
cp /var/tmp/write_splunk.so /usr/lib64/collectd/
mv /etc/collectd.conf "/etc/collectd/collectd.conf.old.$(date +'%Y%m%d-%H%M%S')"
cp /var/tmp/collectd.conf /etc
vi collectd.conf
systemctl enable collectd
systemctl start collectd
